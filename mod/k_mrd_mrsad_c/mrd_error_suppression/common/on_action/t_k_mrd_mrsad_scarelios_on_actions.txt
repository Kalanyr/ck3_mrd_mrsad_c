
MRD_check = {
	effect = {
		if = {
			limit = {
				OR = {
					NOT = { exists = global_var:MRD_is_loaded }
					NOT = { global_var:MRD_is_loaded = yes }
				}
			}
			set_global_variable = {
				name = MRD_is_loaded
				value = yes
			}
		}
	}
}
